1. Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia:

SELECT * FROM `students`
JOIN `degrees`
ON  `students`.`id` = `degree_id`
WHERE `degrees`.`name` = 'Corso di Laurea in Economia';

2. Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di Neuroscienze:

SELECT * FROM `departments`
JOIN `degrees`
ON `departments`.`id` = `degrees`.`department_id`
WHERE `departments`.`name` = 'Dipartimento di Neuroscienze'
AND `degrees`.`level` = 'Magistrale';

3. Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44):

SELECT `courses`.`name` AS 'nome_corso', `teachers`.`name` AS 'nome_insegante',`teachers`.`id` AS 'id_prof'  FROM `teachers`
JOIN `course_teacher`
ON `teachers`.`id` = `course_teacher`.`teacher_id`
JOIN `courses`
ON `courses`.`id` = `course_teacher`.`course_id`
WHERE `teachers`.`id` = 44;

4. Selezionare tutti gli studenti con i dati relativi al corso di laurea a cui sono iscritti e il
relativo dipartimento, in ordine alfabetico per cognome e nome

SELECT  `students`.`surname`,`students`.`name`, `degrees`.`name`, `departments`.`name`  FROM `students`
JOIN `degrees`
ON `degrees`.`id` = `students`.`degree_id`
JOIN `departments`
ON `departments`.`id` = `degrees`.`department_id`
ORDER BY `students`.`surname`, `students`.`name` ASC;


5. Selezionare tutti i corsi di laurea con i relativi corsi e insegnanti

SELECT `degrees`.`name` AS 'corso_di_laurea' , `courses`.`name` AS 'nome_corso_singolo', `teachers`.`name`, `teachers`.`surname` FROM `degrees`
JOIN `courses`
ON `degrees`.`id` = `courses`.`degree_id`
JOIN `course_teacher`
ON `courses`.`id` = `course_teacher`.`course_id`
JOIN `teachers`
ON `teachers`.`id` = `course_teacher`.`teacher_id`
ORDER BY `corso_di_laurea` ASC;

6. Selezionare tutti i docenti che insegnano nel Dipartimento di Matematica (54) // corretto, 54 occorrenze

SELECT DISTINCT `teachers`.`id` AS 'id_insegnante',`teachers`.`name` AS 'nome_insegnante',`teachers`.`surname` AS 'cognome_insegnante', `departments`.`name` AS 'nome_dipartimento' FROM `teachers`
JOIN `course_teacher`
ON `teachers`.`id` = `course_teacher`.`teacher_id`
JOIN `courses`
ON `courses`.`id` = `course_teacher`.`course_id`
JOIN `degrees`
ON `degrees`.`id` = `courses`.`degree_id`
JOIN `departments`
ON `departments`.`id` = `degrees`.`department_id`
WHERE `departments`.`name` = 'Dipartimento di Matematica'
ORDER BY `teachers`.`id` ASC;


7. BONUS: Selezionare per ogni studente quanti tentativi dâ€™esame ha sostenuto per
superare ciascuno dei suoi esami

SELECT  COUNT(`exam_student`.`vote`) AS 'numero_appelli_insuff' ,`students`.`id` AS 'codice_studente' , `students`.`name`,`students`.`surname`, `students`.`registration_number` AS 'matricola' FROM `students`
JOIN `exam_student`
ON `students`.`id` = `exam_student`.`exam_id`
WHERE `exam_student`.`vote` < 18
GROUP BY `students`.`id` ,`exam_student`.`exam_id`;
